var img = document.getElementById("thumbnail"); img.onclick = function() { img.style.width = "500px"; img.style.height = "500px"; img.style.position = "fixed"; img.style.top = "50%"; img.style.left = "50%"; img.style.transform = "translate(-50%, -50%)"; } document.onclick = function(event) { if (event.target != img) { img.style = ""; } }
